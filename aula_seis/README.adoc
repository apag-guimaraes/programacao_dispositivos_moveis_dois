== Google Maps

Através do SDK do Android é possível adicionar mapas ao seu projeto. É necessário criar um projeto na Google Cloud, gerar uma chave de API e adicionar essa chave no seu projeto Android. Para mais detalhes em como criar, configurar e ativar a API do SDK Maps, acesse o material disponível na documentação oficial: 

- GOOGLE DEVELOPERS. Configurar seu projeto do Google Cloud. Disponível em: https://developers.google.com/maps/documentation/android-sdk/get-api-key?hl=pt-br. Acessado em: 26 ago. de 2024.

import android.Manifest;
import android.content.pm.PackageManager;
import android.os.Bundle;
import android.view.View;
import android.widget.FrameLayout;
import android.widget.Toast;
import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.app.ActivityCompat;
import androidx.core.content.ContextCompat;
import org.osmdroid.config.Configuration;
import org.osmdroid.util.GeoPoint;
import org.osmdroid.views.MapView;
import org.osmdroid.views.overlay.Marker;
public class MainActivity extends AppCompatActivity {
    private static final int REQUEST_CODE = 1;
    private final String[] permissoesNecessarias = new String[]{
            Manifest.permission.ACCESS_FINE_LOCATION,
            Manifest.permission.ACCESS_COARSE_LOCATION
    };
    private MapView mapView;
    private FrameLayout mapContainer;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        // Configuração obrigatória do osmdroid antes de carregar o layout
        Configuration.getInstance().setUserAgentValue(getPackageName());
        setContentView(R.layout.activity_main);
        mapContainer = findViewById(R.id.mapContainer);
        solicitarPermissao(permissoesNecessarias);
    }

    private void inicializarMapa() {
        mapView = findViewById(R.id.mapView);
        mapView.setMultiTouchControls(true);
        GeoPoint palmas = new GeoPoint(-10.184, -48.333);
        mapView.getController().setZoom(13.0);
        mapView.getController().setCenter(palmas);
        Marker marker = new Marker(mapView);
        marker.setPosition(palmas);
        marker.setAnchor(Marker.ANCHOR_CENTER, Marker.ANCHOR_BOTTOM);
        marker.setTitle("Palmas - TO");
        marker.setIcon(ContextCompat.getDrawable(this, R.drawable.sunflower));
        mapView.getOverlays().add(marker);
        mapContainer.setVisibility(View.VISIBLE);
    }
    private void solicitarPermissao(String[] permissions) {
        for (String permission : permissions) {
            if (ContextCompat.checkSelfPermission(this, permission)
                    != PackageManager.PERMISSION_GRANTED) {
                ActivityCompat.requestPermissions(this, permissions, REQUEST_CODE);
                return;
            }
        }
        inicializarMapa();
    }
    @Override
    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions,
                                           @NonNull int[] grantResults) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
        if (requestCode == REQUEST_CODE) {
            boolean granted = true;
            // Verifica se alguma permissão foi negada no array de resultados
            if (grantResults.length > 0) {
                for (int result : grantResults) {
                    if (result != PackageManager.PERMISSION_GRANTED) {
                        granted = false;
                        break;
                    }
                }
            } else {
                granted = false;
            }
            if (granted) {
                inicializarMapa();
            } else {
                mapContainer.setVisibility(View.GONE);
                for (String permission : permissoesNecessarias) {
                    if (ActivityCompat.shouldShowRequestPermissionRationale(this, permission)) {
                        mostrarExplicacao = true;
                        break;
                    }
                }
                if (mostrarExplicacao) {
                    // Caso 1: O usuário apenas clicou em "Negar" na primeira tentativa.
                    Toast.makeText(this, 
                            "A permissão de localização é necessária para exibir o mapa corretamente.", 
                            Toast.LENGTH_LONG).show();
                } else {
                    // Caso 2: O usuário marcou "Não perguntar novamente" (Negado permanente).
                    Toast.makeText(this, 
                            "Permissão negada permanentemente. Ative a localização nas configurações.", 
                            Toast.LENGTH_LONG).show();
                    finish(); 
                }
            }
        }
    }
    protected void onResume() {
        super.onResume();
        if (mapView != null && mapContainer.getVisibility() == View.VISIBLE) {
            mapView.onResume();
        }
    }
    @Override
    protected void onPause() {
        super.onPause();
        if (mapView != null && mapContainer.getVisibility() == View.VISIBLE) {
            mapView.onPause();
        }
    }
}

import android.Manifest
import android.content.pm.PackageManager
import android.os.Bundle
import android.view.View
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.ActivityCompat
import androidx.core.content.ContextCompat
import com.example.seuprojeto.databinding.ActivityMainBinding // Ajuste para o pacote do seu app
import org.osmdroid.config.Configuration
import org.osmdroid.util.GeoPoint
import org.osmdroid.views.overlay.Marker
class MainActivity : AppCompatActivity() {
    private lateinit var binding: ActivityMainBinding
    companion object {
        private const val REQUEST_CODE = 1
        // Lista de permissões necessárias para localização
        private val PERMISSOES = arrayOf(
            Manifest.permission.ACCESS_FINE_LOCATION,
            Manifest.permission.ACCESS_COARSE_LOCATION
        )
    }
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        // Configuração obrigatória do osmdroid antes do setContentView
        Configuration.getInstance().userAgentValue = packageName
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)
        verificarESolicitarPermissoes()
    }
    private fun inicializarMapa() {
        // Habilita gestos de pinça para zoom
        binding.mapView.setMultiTouchControls(true)
        // Configura as coordenadas centrais (Palmas - TO) e o nível de zoom
        val palmas = GeoPoint(-10.184, -48.333)
        binding.mapView.controller.setZoom(13.0)
        binding.mapView.controller.setCenter(palmas)
        val marker = Marker(binding.mapView).apply {
            position = palmas
            setAnchor(Marker.ANCHOR_CENTER, Marker.ANCHOR_BOTTOM)
            title = "Palmas - TO"
            icon = ContextCompat.getDrawable(this@MainActivity, R.drawable.sunflower)
        }
        binding.mapView.overlays.add(marker)
        // Torna o container do mapa visível
        binding.mapContainer.visibility = View.VISIBLE
    }
    private fun verificarESolicitarPermissoes() {
        // Verifica se alguma das permissões ainda não foi concedida
        val precisaSolicitar = PERMISSOES.any {
            ContextCompat.checkSelfPermission(this, it) != PackageManager.PERMISSION_GRANTED
        }
        if (precisaSolicitar) {
            ActivityCompat.requestPermissions(this, PERMISSAO_ARRAY, REQUEST_CODE)
        } else {
            inicializarMapa()
        }
    }
    override fun onRequestPermissionsResult(
        requestCode: Int,
        permissions: Array<out String>,
        grantResults: IntArray
    ) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults)
        if (requestCode == REQUEST_CODE) {
            // Verifica se todas as permissões no array foram aceitas
            val todasConcedidas = grantResults.isNotEmpty() && grantResults.all { 
                it == PackageManager.PERMISSION_GRANTED 
            }
            if (todasConcedidas) {
                inicializarMapa()
            } else {
                binding.mapContainer.visibility = View.GONE
                val mostrarExplicacao = PERMISSOES.any {
                    ActivityCompat.shouldShowRequestPermissionRationale(this, it)
                }
                if (mostrarExplicacao) {
                    // Caso 1: Usuário clicou em "Negar" pela primeira vez.
                    Toast.makeText(
                        this, 
                        "A localização é necessária para centralizar o mapa na sua região.", 
                        Toast.LENGTH_LONG
                    ).show()
                    
                    // Aqui você também poderia alternativamente disparar o seu custom DialogFragment
                } else {
                    // Caso 2: O usuário marcou "Não perguntar novamente" ou negou de forma permanente.
                    Toast.makeText(
                        this, 
                        "Permissão negada permanentemente. Ative nas configurações do sistema.", 
                        Toast.LENGTH_LONG
                    ).show()
                    finish()
                }
            }
        }
    }
    override fun onResume() {
        super.onResume()
        if (binding.mapContainer.visibility == View.VISIBLE) {
            binding.mapView.onResume()
        }
    }
    override fun onPause() {
        super.onPause()
        if (binding.mapContainer.visibility == View.VISIBLE) {
            binding.mapView.onPause()
        }
    }
}

            return;

        Location lastLocation = locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER);
        if (lastLocation != null) {
            inicializarMapa(lastLocation);

            locationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER, 2000, 10, new LocationListener() {
                @Override

                    inicializarMapa(location);
                    locationManager.removeUpdates(this);

                return;
            }

        iniciarLocalizacaoTempoReal();

    private void iniciarLocalizacaoTempoReal() {
        locationManager = (LocationManager) getSystemService(LOCATION_SERVICE);
        if (ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED
                && ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
            return;
        }
        Location lastLocation = locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER);
        if (lastLocation != null) {
            inicializarMapa(lastLocation);
        }
        locationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER,
                2000, 
                5,    
                new LocationListener() {
                    @Override
                    public void onLocationChanged(@NonNull Location location) {
                        GeoPoint novaPosicao = new GeoPoint(location.getLatitude(), location.getLongitude());
                        marker.setPosition(novaPosicao);
                        mapView.getController().animateTo(novaPosicao);
                        txtCoordenadas.setText("Coordenadas: " +

        mapView.getController().setZoom(16.0);

        // Cria o marcador na posição inicial detectada
        marker = new Marker(mapView);
        marker.setPosition(pontoInicial);
        marker.setAnchor(Marker.ANCHOR_CENTER, Marker.ANCHOR_BOTTOM);
        marker.setTitle("Minha localização");

    private void configurarCallbackLocalizacao() {
        locationCallback = new LocationCallback() {
            @Override
            public void onLocationResult(@NonNull LocationResult locationResult) {
                for (Location location : locationResult.getLocations()) {
                    if (location != null) {
                        GeoPoint novaPosicao = new GeoPoint(location.getLatitude(), location.getLongitude());
                        
                        // Segurança: se o mapa ainda não foi inflado pela última localização conhecida, cria agora
                        if (mapView == null) {
                            inicializarMapa(location);
                        } else {
                            if (marker != null) {
                                marker.setPosition(novaPosicao);
                            }
                            mapView.getController().animateTo(novaPosicao);
                            txtCoordenadas.setText("Coordenadas: " + novaPosicao.getLatitude() + ", " + novaPosicao.getLongitude());
                        }
                    }
                }
            }
        };
    }

        for (String permission : permissions) {
            if (ContextCompat.checkSelfPermission(this, permission) != PackageManager.PERMISSION_GRANTED) {
                ActivityCompat.requestPermissions(this, permissions, REQUEST_CODE);
                return;
            }
        }
        iniciarLocalizacaoTempoReal();

        if (ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED 
                && ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
            return;

        // Passo 1: Tenta ler o cache rápido de localização
        fusedLocationClient.getLastLocation().addOnSuccessListener(this, new OnSuccessListener<Location>() {
            @Override
            public void onSuccess(Location location) {
                if (location != null) {
                    inicializarMapa(location);
                }
            }
        });
        // Passo 2: Cria a requisição contínua (Equivalente ao requestLocationUpdates do LocationManager)
        LocationRequest locationRequest = new LocationRequest.Builder(Priority.PRIORITY_HIGH_ACCURACY, 2000)
                .setMinUpdateDistanceMeters(5) // Atualiza se deslocar 5 metros
                .build();
        // Vincula as atualizações ao ciclo do Looper principal do app
        fusedLocationClient.requestLocationUpdates(locationRequest, locationCallback, Looper.getMainLooper());
    }
    @Override

        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
        if (requestCode == REQUEST_CODE) {
            boolean granted = true;
            if (grantResults.length > 0) {
                for (int result : grantResults) {
                    if (result != PackageManager.PERMISSION_GRANTED) {
                        granted = false;
                        break;
                    }
                }
            } else {
                granted = false;
            }
            if (granted) {
                iniciarLocalizacaoTempoReal();
            } else {
                mapContainer.setVisibility(View.GONE);
                // ADICIONADO: Avaliação do Rationale para exibição do diálogo explicativo
                boolean mostrarExplicacao = false;
                for (String permission : permissoesNecessarias) {
                    if (ActivityCompat.shouldShowRequestPermissionRationale(this, permission)) {
                        mostrarExplicacao = true;
                        break;
                    }
                }
                if (mostrarExplicacao) {
                    // Caso 1: Usuário apenas clicou em "Negar" pela primeira vez
                    Toast.makeText(this, 
                            "A permissão de localização é fundamental para acompanhar o seu deslocamento no mapa.", 
                            Toast.LENGTH_LONG).show();
                } else {
                    // Caso 2: Marcado permanentemente como "Não perguntar novamente"
                    Toast.makeText(this, 
                            "Funcionalidade desabilitada permanentemente. Permita o acesso nas configurações do celular.", 
                            Toast.LENGTH_LONG).show();
                    finish();
                }
            }
        }
    }

        super.onResume();
        if (mapView != null && mapContainer.getVisibility() == View.VISIBLE) {

    }
    @Override
    protected void onPause() {
        super.onPause();
        if (mapView != null) {
            mapView.onPause();
        }
        // Desativa a escuta para economizar bateria enquanto o app está em segundo plano
        if (fusedLocationClient != null && locationCallback != null) {
            fusedLocationClient.removeLocationUpdates(locationCallback);
        }
    }

import com.example.seuprojeto.databinding.ActivityMainBinding // Ajuste para o pacote correto do seu app
import com.google.android.gms.location.*
import org.osmdroid.config.Configuration
import org.osmdroid.util.GeoPoint
import org.osmdroid.views.overlay.Marker
class MainActivity : AppCompatActivity() {
    private lateinit var binding: ActivityMainBinding
    private var marker: Marker? = null
    // Componentes do Fused Location
    private lateinit var fusedLocationClient: FusedLocationProviderClient
    private lateinit var locationCallback: LocationCallback
    companion object {
        private const val REQUEST_CODE = 1
        private val PERMISSOES_NECESSARIAS = arrayOf(
            Manifest.permission.ACCESS_FINE_LOCATION,
            Manifest.permission.ACCESS_COARSE_LOCATION
        )
    }
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        // Configuração obrigatória do osmdroid antes de inflar o layout
        Configuration.getInstance().userAgentValue = packageName
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)
        // Inicializa o cliente do Google
        fusedLocationClient = LocationServices.getFusedLocationProviderClient(this)
        // Configura a escuta de atualizações contínuas
        configurarCallbackLocalizacao()
        verificarESolicitarPermissoes()
    }
    private fun inicializarMapa(location: Location) {
        binding.mapView.setMultiTouchControls(true)
        val pontoInicial = GeoPoint(location.latitude, location.longitude)
        binding.mapView.controller.setZoom(16.0)
        binding.mapView.controller.setCenter(pontoInicial)
        // Criação e configuração fluida do Marcador com 'apply'
        marker = Marker(binding.mapView).apply {
            position = pontoInicial
            setAnchor(Marker.ANCHOR_CENTER, Marker.ANCHOR_BOTTOM)
            title = "Minha localização"
        }
        binding.mapView.overlays.add(marker)
        binding.mapContainer.visibility = View.VISIBLE
        binding.txtCoordenadas.text = "Coordenadas: ${pontoInicial.latitude}, ${pontoInicial.longitude}"
    }
    private fun configurarCallbackLocalizacao() {
        locationCallback = object : LocationCallback() {
            override fun onLocationResult(locationResult: LocationResult) {
                for (location in locationResult.locations) {
                    location?.let {
                        val novaPosicao = GeoPoint(it.latitude, it.longitude)
                        // Inicializa o mapa caso a leitura do cache rápido (LastLocation) tenha falhado
                        if (binding.mapView.overlays.isEmpty()) {
                            inicializarMapa(it)
                        } else {
                            marker?.position = novaPosicao
                            binding.mapView.controller.animateTo(novaPosicao)
                            binding.txtCoordenadas.text = "Coordenadas: ${novaPosicao.latitude}, ${novaPosicao.longitude}"
                        }
                    }
                }
            }
        }
    }
    private fun verificarESolicitarPermissoes() {
        // Verifica se há alguma permissão pendente de autorização
        val precisaSolicitar = PERMISSOES_NECESSARIAS.any {
            ContextCompat.checkSelfPermission(this, it) != PackageManager.PERMISSION_GRANTED
        }
        if (precisaSolicitar) {
            ActivityCompat.requestPermissions(this, PERMISSOES_NECESSARIAS, REQUEST_CODE)
        } else {
            iniciarLocalizacaoTempoReal()
        }
    }
    private fun iniciarLocalizacaoTempoReal() {
        if (ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED &&
            ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED
        ) {
            return
        }
        // 1. Tenta obter o cache rápido de posicionamento
        fusedLocationClient.lastLocation.addOnSuccessListener(this) { location: Location? ->
            location?.let { inicializarMapa(it) }
        }
        // 2. Monta a requisição contínua de alta precisão
        val locationRequest = LocationRequest.Builder(Priority.PRIORITY_HIGH_ACCURACY, 2000L)
            .setMinUpdateDistanceMeters(5f)
            .build()
        // Registra o monitoramento em tempo real
        fusedLocationClient.requestLocationUpdates(locationRequest, locationCallback, Looper.getMainLooper())
    }
    override fun onRequestPermissionsResult(
        requestCode: Int,
        permissions: Array<out String>,
        grantResults: IntArray
    ) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults)
        if (requestCode == REQUEST_CODE) {
            val todasConcedidas = grantResults.isNotEmpty() && grantResults.all { 
                it == PackageManager.PERMISSION_GRANTED 
            }
            if (todasConcedidas) {
                iniciarLocalizacaoTempoReal()
            } else {
                binding.mapContainer.visibility = View.GONE
                // Tratamento simplificado com Rationale usando a extensão 'any'
                val mostrarExplicacao = PERMISSOES_NECESSARIAS.any {
                    ActivityCompat.shouldShowRequestPermissionRationale(this, it)
                }

                        this,
                        "A permissão de localização é fundamental para acompanhar o seu deslocamento no mapa.",
                        Toast.LENGTH_LONG
                    ).show()

                        this,
                        "Funcionalidade desabilitada permanentemente. Permita o acesso nas configurações do celular.",

                    ).show()

                }

        }
