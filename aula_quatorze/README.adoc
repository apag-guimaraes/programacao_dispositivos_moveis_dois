== Configurações básicas

Inicialmente deve ser criado um projeto no Firebase console e depois adicioná-los em seu projeto no Android Studio. A documentação oficial explica esse processo passo a passo. Dessa forma, acesse: https://firebase.google.com/docs/android/setup?hl=pt-br.

== Conexão com o banco

== Salvar / listar dados no banco

Para salvar os dados no Firestore, você precisa criar uma coleção e os dados serão salvos na estrutura de um Map.

//código para salvar dados no Firestore

   db.collection("minha lista")
                .add(lista.toMap())
                .addOnSuccessListener(documentReference -> {

                    obterDados();
                })
                .addOnFailureListener(e -> {
                    Log.w("Cadastro", "Erro", e);

Para obter os dados do Firestore basta acessar QueryDocumentSnapshot document : task.getResult() e através do document acessar os campos dos objetos mapeados, por exemplo document.getString("nome do campo"), no caso de inteiro document.getLong("nome do campo"). Veja o código completo abaixo:

 db.collection("minha lista")
                .get()
                .addOnCompleteListener(task -> {
                    if (task.isSuccessful()) {
                        List<String> lista = new ArrayList<>();
                        for (QueryDocumentSnapshot document : task.getResult()) {
                            String nome = document.getString("nome do campo");
                            int idade = Objects.requireNonNull(document.getLong("nome do campo")).intValue();
                            Log.d("Dado",document.getId() + " => " + document.getData());
                        }  
                    } else {
                        Log.d("Erro", "Erro ao obter dados: ", task.getException());
                    }
                });

Para mais detalhes sobre leitura e gravação de dados no Realtime Database acesse a documentação oficial: https://firebase.google.com/docs/database/android/read-and-write?hl=pt-br#java_2.

import android.os.Bundle;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;
import androidx.lifecycle.ViewModelProvider;
import androidx.recyclerview.widget.LinearLayoutManager;
import com.example.seuprojeto.databinding.ActivityMainBinding; // Importe o binding gerado
public class MainActivity extends AppCompatActivity {
    private ActivityMainBinding binding; // Referência ao binding
    private UsuarioViewModel viewModel;
    private UsuarioAdapter adapter;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        // Inicializa o ViewBinding
        binding = ActivityMainBinding.inflate(getLayoutInflater());
        setContentView(binding.getRoot());
        binding.recyclerUsuarios.setLayoutManager(new LinearLayoutManager(this));
        viewModel = new ViewModelProvider(this).get(UsuarioViewModel.class);
        // Observer para a lista
        viewModel.getUsuarios().observe(this, usuarios -> {
            adapter = new UsuarioAdapter(usuarios);
            binding.recyclerUsuarios.setAdapter(adapter);
        });
        // Observer para erros
        viewModel.getErro().observe(this, error -> 
            Toast.makeText(this, error, Toast.LENGTH_SHORT).show()
        );
        // Click Listener do botão usando o binding
        binding.buttonConectar.setOnClickListener(v -> {
            Usuario usuario = new Usuario(null, "Ana", "email@email.com");
            viewModel.addUsuario(usuario);
        });
    }
}

- UsuarioRepository.java
[source,java]
import androidx.annotation.NonNull;
import com.google.firebase.database.*;
public class UsuarioRepository {
    private final DatabaseReference firebase;
    private ValueEventListener valueEventListener; // Referência para remover depois
    public UsuarioRepository() {
        firebase = FirebaseDatabase.getInstance().getReference("usuarios");
    }
    public void addUsuario(Usuario usuario) {
        String id = firebase.push().getKey();
        if (id != null) {
            usuario.setId(id);
            firebase.child(id).setValue(usuario);
        }
    }
    public void obterUsuarios(UsuariosListener listener) {
        // Remove listener anterior se existir para evitar duplicação
        if (valueEventListener != null) firebase.removeEventListener(valueEventListener);
        valueEventListener = new ValueEventListener() {
            @Override
            public void onDataChange(@NonNull DataSnapshot snapshot) {
                List<Usuario> list = new ArrayList<>();
                for (DataSnapshot child : snapshot.getChildren()) {
                    Usuario usuario = child.getValue(Usuario.class);
                    if (usuario != null) list.add(usuario);
                }
                listener.onSuccess(list);
            }
            @Override
            public void onCancelled(@NonNull DatabaseError error) {
                listener.onError(error.getMessage());
            }
        };
        firebase.addValueEventListener(valueEventListener);
    }
    // Método importante para chamar no onCleared do ViewModel
    public void removerListener() {
        if (valueEventListener != null) {
            firebase.removeEventListener(valueEventListener);
        }
    }
}

import androidx.lifecycle.LiveData;
import androidx.lifecycle.MutableLiveData;
import androidx.lifecycle.ViewModel;
import java.util.List;
public class UsuarioViewModel extends ViewModel {
    private UsuarioRepository repository = new UsuarioRepository();
    private MutableLiveData<List<Usuario>> usuariosLiveData = new MutableLiveData<>();
    private MutableLiveData<String> erroLiveData = new MutableLiveData<>();
    public UsuarioViewModel() {
        obterUsuarios();
    }
    public LiveData<List<Usuario>> getUsuarios() {
        return usuariosLiveData;
    }
    public LiveData<String> getErro() {
        return erroLiveData;
    }
    public void obterUsuarios() {
        repository.obterUsuarios(new UsuarioRepository.UsuariosListener() {
            @Override
            public void onSuccess(List<Usuario> usuarios) {
                usuariosLiveData.setValue(usuarios);
            }
            @Override
            public void onError(String error) {
                erroLiveData.setValue(error);
            }
        });
    }
    public void addUsuario(Usuario usuario) {
        repository.addUsuario(usuario);
    }
    public void deleteUsuario(String id) {
        repository.deleteUser(id);
    }
    public void updateUsuario(Usuario usuario) {
        repository.updateUser(usuario);
    }
}

import com.google.firebase.database.*
class UsuarioRepository {
    private val firebase = FirebaseDatabase.getInstance().getReference("usuarios")
    private var valueEventListener: ValueEventListener? = null
    fun addUsuario(usuario: Usuario) {
        val id = firebase.push().key
        if (id != null) {
            usuario.id = id
            firebase.child(id).setValue(usuario)
        }
    }
    fun obterUsuarios(onSuccess: (List<Usuario>) -> Unit, onError: (String) -> Unit) {
        valueEventListener?.let { firebase.removeEventListener(it) }
        valueEventListener = object : ValueEventListener {
            override fun onDataChange(snapshot: DataSnapshot) {
                val list = snapshot.children.mapNotNull { it.getValue(Usuario::class.java) }
                onSuccess(list)
            }
            override fun onCancelled(error: DatabaseError) {
                onError(error.message)
            }
        }.also { firebase.addValueEventListener(it) }
    }
    fun removerListener() {
        valueEventListener?.let { firebase.removeEventListener(it) }
    }
}

import androidx.lifecycle.*
class UsuarioViewModel : ViewModel() {
    private val repository = UsuarioRepository()
    private val _usuariosLiveData = MutableLiveData<List<Usuario>>()
    val usuariosLiveData: LiveData<List<Usuario>> get() = _usuariosLiveData
    private val _erroLiveData = MutableLiveData<String>()
    val erroLiveData: LiveData<String> get() = _erroLiveData
    init { obterUsuarios() }
    private fun obterUsuarios() {
        repository.obterUsuarios(
            onSuccess = { _usuariosLiveData.value = it },
            onError = { _erroLiveData.value = it }
        )
    }
    fun addUsuario(usuario: Usuario) = repository.addUsuario(usuario)
    override fun onCleared() {
        super.onCleared()
        repository.removerListener()
    }
}
