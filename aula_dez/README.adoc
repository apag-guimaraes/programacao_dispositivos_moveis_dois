= Handler

Permite a troca de mensagens entre a pilha de execução principal e uma pilha de execução secundária. Lembre-se a função da thread principal é redenrizar 
a interface gráfica. Se há a necessidade de fazer uma conexão remota, isto deve ficar a cargo da thread secundária, se um determinado valor presente na 
thread secundária é necessário para atualizar uma view da interface gráfica, o handler pode ser essa ponte entre as duas threads.

Para mais detalhes sobre handler: GOOGLE DEVELOPERS. Handler. Disponível em: https://developer.android.com/reference/android/os/Handler. Acesso em:
29 ago. 2025.

                    Thread.sleep(2000);

                });
            }

        }).start();

import android.os.Bundle
import android.os.Handler
import android.os.Looper
import android.util.Log
import androidx.appcompat.app.AppCompatActivity
import com.seu.pacote.databinding.ActivityMainBinding // Ajuste para o seu package
import kotlin.random.Random
class MainActivity : AppCompatActivity() {
    private lateinit var binding: ActivityMainBinding
    private val handler = Handler(Looper.getMainLooper())
    private var numberG: Int = 0
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        // Inflar o layout com ViewBinding
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)
        binding.buttonClicar.setOnClickListener {
            iniciarProcessamento()
        }
    }
    private fun iniciarProcessamento() {
        binding.buttonClicar.isEnabled = false
        Log.i("Main", "Thread atual: ${Thread.currentThread().name}")
        // Iniciar thread em background
        Thread {
            for (i in 0 until 10) {
                Log.i("Thread", "Thread em execução: ${Thread.currentThread().name}")
                numberG = Random.nextInt(21)
                try {
                    Thread.sleep(2000)
                } catch (e: InterruptedException) {
                    Log.e("Thread", "Erro: ${e.message}")
                }
                handler.post {
                    Log.i("Post", "Thread que atualiza UI: ${Thread.currentThread().name}")
                    binding.textViewNumber.text = numberG.toString()
                }
            }
            handler.post {
                binding.buttonClicar.isEnabled = true
            }
        }.start()
    }
}

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.os.Handler;
import android.os.Looper;
import android.os.Message;
import android.util.Log;
import com.seu.pacote.databinding.ActivityMainBinding; // Ajuste para o seu package
import java.util.Random;
public class MainActivity extends AppCompatActivity {
    private ActivityMainBinding binding;
    private Handler handler;
    private static final String TAG = "Thread-MAIN";
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        // Inicializa o ViewBinding
        binding = ActivityMainBinding.inflate(getLayoutInflater());
        setContentView(binding.getRoot());
        // Inicializa o Handler
        handler = new Handler(Looper.getMainLooper()) {
            @Override
            public void handleMessage(@NonNull Message msg) {
                // Acesso via binding
                binding.textViewNumber.setText(String.valueOf(msg.arg1));
                Log.i(TAG, "UI atualizada na thread: " + Thread.currentThread());
            }
        };
        // Define o clique de forma programática
        binding.buttonClicar.setOnClickListener(v -> clicar());
    }
    private void clicar() {
        binding.buttonClicar.setEnabled(false);
        new Thread(() -> {
            Random random = new Random();
            for (int i = 0; i < 10; i++) {
                int num = random.nextInt(21);
                // Envia a mensagem para o Handler
                Message msg = handler.obtainMessage();
                msg.arg1 = num;
                handler.sendMessage(msg);
                try {
                    Thread.sleep(2000);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }
            handler.post(() -> binding.buttonClicar.setEnabled(true));
        }).start();
    }
    @Override
    protected void onDestroy() {
        super.onDestroy();
        // Remove mensagens pendentes para evitar memory leaks
        handler.removeCallbacksAndMessages(null);
    }
}

import android.os.Bundle
import android.os.Handler
import android.os.Looper
import android.os.Message
import android.util.Log
import androidx.appcompat.app.AppCompatActivity
import com.seu.pacote.databinding.ActivityMainBinding // Ajuste para o seu package
import kotlin.random.Random
class MainActivity : AppCompatActivity() {
    private lateinit var binding: ActivityMainBinding
    private lateinit var handler: Handler
    companion object {
        private const val TAG = "Thread-MAIN"
    }
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        // Inicializa o ViewBinding
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)
        // Inicializa o Handler
        handler = object : Handler(Looper.getMainLooper()) {
            override fun handleMessage(msg: Message) {
                // Acesso via binding
                binding.textViewNumber.text = msg.arg1.toString()
                Log.i(TAG, "UI atualizada na thread: ${Thread.currentThread()}")
            }
        }
        // Define o clique de forma programática
        binding.buttonClicar.setOnClickListener { clicar() }
    }
    private fun clicar() {
        binding.buttonClicar.isEnabled = false
        Thread {
            val random = Random(System.currentTimeMillis())
            for (i in 0 until 10) {
                val num = random.nextInt(21)
                // Envia a mensagem para o Handler
                val msg = handler.obtainMessage()
                msg.arg1 = num
                handler.sendMessage(msg)
                try {
                    Thread.sleep(2000)
                } catch (e: InterruptedException) {
                    Thread.currentThread().interrupt()
                }
            }
            // Reabilita o botão na thread principal
            handler.post { binding.buttonClicar.isEnabled = true }
        }.start()
    }
    override fun onDestroy() {
        super.onDestroy()
        // Remove mensagens pendentes para evitar memory leaks
        handler.removeCallbacksAndMessages(null)
    }
}
